import { useGSAP } from '@gsap/react'
import gsap from 'gsap'
import React, { useContext, useRef, useEffect } from 'react'
import { NavbarContext } from '../../context/NavContext'

const FullScreenNav = () => {
  const fullNavLinkRef = useRef(null)
  const fullScreenRef = useRef(null)
  const [navOpen, setnavOpen] = useContext(NavbarContext)

  const tl = useRef()

  useGSAP(() => {
    // create timeline once
    tl.current = gsap.timeline({ paused: true })

    // initial hidden state
    tl.current.set(fullScreenRef.current, { autoAlpha: 0, display: "none" })

    // show overlay
    tl.current.to(fullScreenRef.current, { autoAlpha: 1, display: "block", duration: 0.3 })

    // stairing animation
    tl.current.from(".stairing", {
      height: 0,
      stagger: { amount: 0.3 },
      duration: 0.5,
      ease: "power2.out"
    })

    // links animation
    tl.current.from(".link", {
      opacity: 0,
      rotateX: 90,
      transformOrigin: "top",
      stagger: { amount: 0.2 },
      duration: 0.4,
      ease: "back.out(1.7)"
    })

    // nav top (logo/close btn)
    tl.current.from(".navLink", {
      opacity: 0,
      y: -20,
      duration: 0.3
    })
  }, [])

  useEffect(() => {
    if (navOpen) {
      tl.current.play()
    } else {
      tl.current.reverse()
    }
  }, [navOpen])

  return (
    <div 
      ref={fullScreenRef} 
      id='fullscreennav' 
      className='fixed h-screen w-full bg-black overflow-x-hidden z-20'
    >
      <div className='h-screen w-full fixed'>
        <div className='h-full w-full flex'>
          <div className='stairing h-full w-1/5 bg-red-500'></div>
          <div className='stairing h-full w-1/5 bg-red-500'></div>
          <div className='stairing h-full w-1/5 bg-red-500'></div>
          <div className='stairing h-full w-1/5 bg-red-500'></div>
          <div className='stairing h-full w-1/5 bg-red-500'></div>
        </div>
      </div>
      <div ref={fullNavLinkRef} className='relative'>
        <div className='navLink flex w-full justify-between items-start'>
          <div className='p-5'>
            <div className='w-30'>
              <svg className='w-full' xmlns="http://www.w3.org/2000/svg" viewBox="0 0 103 44">
                <path fill='white' fillRule="evenodd" d="M35.1441047,8.4486911 L58.6905011,8.4486911 L58.6905011,-1.3094819e-14 L35.1441047,-1.3094819e-14 L35.1441047,8.4486911 Z M20.0019577,0.000230366492 L8.83414254,25.3433089 L18.4876971,25.3433089 L29.5733875,0.000230366492 L20.0019577,0.000230366492 Z M72.5255345,0.000691099476 L72.5255345,8.44846073 L94.3991559,8.44846073 L94.3991559,16.8932356 L72.5275991,16.8932356 L72.5275991,19.5237906 L72.5255345,19.5237906 L72.5255345,43.9274346 L102.80937,43.9274346 L102.80937,35.4798953 L80.9357483,35.4798953 L80.9357483,25.3437696 L94.3996147,25.3428482 L94.3996147,16.8953089 L102.80937,16.8953089 L102.80937,0.000691099476 L72.5255345,0.000691099476 Z M-1.30398043e-14,43.9278953 L8.78642762,43.9278953 L8.78642762,0.0057591623 L-1.30398043e-14,0.0057591623 L-1.30398043e-14,43.9278953 Z M58.6849955,8.4486911 L43.1186904,43.9274346 L52.3166592,43.9274346 L67.9877996,8.4486911 L58.6849955,8.4486911 Z M18.4688864,25.3437696 L26.7045278,43.9278953 L36.2761871,43.9278953 L28.1676325,25.3375497 L18.4688864,25.3437696 Z"></path>
              </svg>
            </div>
          </div>
          <div 
            onClick={() => setnavOpen(false)} 
            className='relative mt-5 mr-5 w-20 h-20 cursor-pointer'
          >
            <div className='h-28 w-0.5 -rotate-45 origin-top absolute bg-white rounded-full'></div>
            <div className='h-28 w-0.5 right-0 rotate-45 origin-top absolute bg-white rounded-full'></div>
          </div>
        </div>

        {/* Links */}
        <div id='all-links' className='mt-16'>
          <div className='link origin-top relative border-t-2 border-white '>
            <h1 className='font-[font2] text-[7vw] text-center leading-[1] mt-3'>WORK</h1>
          </div>
          <div className='link origin-top relative border-t-2 border-white '>
            <h1 className='font-[font2] text-[7vw] text-center leading-[1] mt-3'>AGENCY</h1>
          </div>
          <div className='link origin-top relative border-t-2 border-white '>
            <h1 className='font-[font2] text-[7vw] text-center leading-[1] mt-3'>CONTACT</h1>
          </div>
          <div className='link origin-top relative border-y-2 border-white '>
            <h1 className='font-[font2] text-[7vw] text-center leading-[1] mt-3'>BLOG</h1>
          </div>
        </div>
      </div>
    </div>
  )
}

export default FullScreenNav
